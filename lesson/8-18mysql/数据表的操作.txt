简单语法
create table [表名]（
  [字段名] ， [数据类型（长度）]，
  [字段名]，[数据类型（长度）]

）

包含约束的

约束 ：主键
       外键
       非空
       唯一
       默认
       自增
1,主键
create table [name](
  id int(10) primary key,
  name varchar(20)
)
create table [name](
  id int(10) ,
  name varchar(20),
  primary key (id,name)
)

2,非空 唯一 默认
create table [name](
      id int(10)         primary key,
      name varchar(20)   not null,
      age int(3) unique  default 10,
      class varchar(10)  default 'php'
)
 3,自增 ai
 auto_increment
 create table [name](
  id int(10) primary key auto_increament,
  name varchar(20)
)

4,外键约束
 student 表
